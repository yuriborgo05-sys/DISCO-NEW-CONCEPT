rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // --- FUNZIONI DI SUPPORTO ---
    function signedIn() {
      return request.auth != null;
    }

    function hasRole(roles) {
      return signedIn() && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in roles;
    }

    function isAdmin() {
      return hasRole(['admin', 'regia', 'direzione']);
    }

    // --- REGOLE COLLEZIONI ---

    // 1. USERS: Lettura/Scrittura permessa solo al proprietario o agli Admin/Head PR
    match /users/{userId} {
      allow read: if signedIn() && (request.auth.uid == userId || hasRole(['admin', 'head_pr', 'pr', 'regia']));
      allow write: if signedIn() && (request.auth.uid == userId || isAdmin());
    }
    
    // 2. ORDERS: Clienti creano/leggono i propri. Staff legge e aggiorna tutto
    match /orders/{orderId} {
      allow create: if signedIn();
      allow read: if signedIn() && (resource.data.userId == request.auth.uid || hasRole(['admin', 'head_pr', 'waiter', 'pr', 'cashier', 'regia']));
      allow update: if signedIn() && hasRole(['admin', 'waiter', 'cashier', 'regia']);
    }
    
    // 3. STAFF MESSAGES: Solo per personale operativo
    match /staffMessages/{msgId} {
      allow read, write: if signedIn() && hasRole(['admin', 'head_pr', 'pr', 'image_girl', 'photographer', 'waiter', 'regia']);
    }

    // 4. SILENT CHAT: Tutti gli autenticati possono scrivere (socializzare) e leggere
    match /silentMessages/{msgId} {
      allow read, write: if signedIn();
    }

    // 5. SERVICE CALLS (Chiamate Cameriere): Cliente crea, Staff gestisce
    match /serviceCalls/{callId} {
      allow create: if signedIn();
      allow read, update: if signedIn() && hasRole(['admin', 'waiter', 'regia']);
    }

    // 6. REVIEWS: Tutti leggono, solo Admin cancella
    match /reviews/{revId} {
      allow read, create: if signedIn();
      allow delete: if isAdmin();
    }

    // 7. SYSTEM STATE (Hacker Mode / Strobe): Lettura globale, Scrittura Admin
    match /system/{state} {
      allow read: if signedIn();
      allow write: if isAdmin();
    }

    // 8. RESERVATIONS: Cliente crea/legge le sue, Staff gestisce
    match /reservations/{resId} {
      allow create: if signedIn();
      allow read: if signedIn() && (resource.data.userId == request.auth.uid || hasRole(['admin', 'head_pr', 'pr', 'regia']));
      allow update: if signedIn() && hasRole(['admin', 'head_pr', 'regia']);
    }
  }
}
